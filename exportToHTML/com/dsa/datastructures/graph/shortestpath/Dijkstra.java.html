66       static void createGraph() 
67       { 
68           adj = new ArrayList[nodes]; 
69    
70           for (int i = 0; i < nodes; i++) 
71               adj[i] = new ArrayList<>(); 
72    
73           for (int i = 0; i < edges; i++) 
74           { 
75               int from, to, weight; 
76    
77               from = in.nextInt() - 1; 
78               to = in.nextInt() - 1; 
79               weight = in.nextInt(); 
80               adj[from].add(new Edge(to, weight)); 
81               adj[to].add(new Edge(from, weight)); 
82           }

85       static void findShortestPaths(int start) 
86       { 
87           distance = new int[nodes]; 
88    
89           for (int i = 0; i < nodes; i++) 
90               distance[i] = INFINITY; 
91    
92           TreeSet<Node> set = new TreeSet<>(); 
93    
94           distance[start] = 0; 
95           set.add(new Node(start, 0, -1)); 
96    
97           while (set.size() > 0) 
98           { 
99               Node curr = set.pollFirst(); 
100   
101              for (Edge edge : adj[curr.node]) 
102              { 
103                  if (distance[curr.node] + edge.weight < distance[edge.to]) 
104                  { 
105                      set.remove(new Node(edge.to, distance[edge.to], -1)); 
106                      distance[edge.to] = distance[curr.node] + edge.weight; 
107                      set.add(new Node(edge.to, distance[edge.to], curr.node)); 
108                  } 
109              } 
110          } 
111      }

158          public int read() 
159          { 
160              if (numChars == -1) 
161                  throw new InputMismatchException(); 
162   
163              if (curChar >= numChars) 
164              { 
165                  curChar = 0; 
166                  try 
167                  { 
168                      numChars = stream.read(buf); 
169                  } 
170                  catch (IOException e) 
171                  { 
172                      throw new InputMismatchException(); 
173                  } 
174                  if (numChars <= 0) 
175                      return -1; 
176              } 
177   
178              return buf[curChar++]; 
179          }

181          public int nextInt() 
182          { 
183              int c = read(); 
184   
185              while (isSpaceChar(c)) 
186                  c = read(); 
187   
188              int sgn = 1; 
189   
190              if (c == '-') 
191              { 
192                  sgn = -1; 
193                  c = read(); 
194              } 
195   
196              int res = 0; 
197   
198              do 
199              { 
200                  if (c < '0' || c > '9') 
201                      throw new InputMismatchException(); 
202   
203                  res *= 10; 
204                  res += c & 15; 
205   
206                  c = read(); 
207              } while (!isSpaceChar(c)); 
208   
209              return res * sgn; 
210          }
